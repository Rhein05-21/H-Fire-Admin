-- 1. Create the 'devices' table to map MAC addresses to Households
CREATE TABLE IF NOT EXISTS public.devices (
    mac TEXT PRIMARY KEY,
    house_name TEXT NOT NULL,
    community TEXT, -- Added community column
    label TEXT NOT NULL,
    profile_id TEXT, 
    created_at TIMESTAMPTZ DEFAULT NOW(),
    last_seen TIMESTAMPTZ
);

-- 4. NEW: Create the 'incidents' table for Active Alarm Tracking
-- This table is for the Admin to "Resolve" alarms.
CREATE TABLE IF NOT EXISTS public.incidents (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    device_mac TEXT NOT NULL REFERENCES public.devices(mac),
    status TEXT NOT NULL DEFAULT 'Active', -- 'Active' or 'Resolved'
    start_time TIMESTAMPTZ DEFAULT NOW(),
    end_time TIMESTAMPTZ,
    resolved_by TEXT REFERENCES public.profiles(id),
    notes TEXT,
    ppm_at_trigger INTEGER
);

-- 5. Enable Realtime for the 'incidents' table
-- This triggers the Siren in the Admin app immediately!
ALTER PUBLICATION supabase_realtime ADD TABLE incidents;
